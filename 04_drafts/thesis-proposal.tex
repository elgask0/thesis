\subsubsection{Event-Study Model Specification}
% The Sun--Abraham estimating equation:

\begin{equation}
Y_{p,t} = \alpha_p + \delta_t
+ \sum_{\ell \in \mathcal{L},\, \ell \neq -1}
  \beta_\ell \, \mathbb{1}\{t - E_p = \ell\} \cdot \mathbb{1}\{p \in \mathcal{G}\}
+ \mathbf{X}_{p,t}' \boldsymbol{\gamma}
+ \varepsilon_{p,t}
\label{eq:event_study}
\end{equation}

% DEFINITIONS:
% - Y_{p,t}: monthly outcome in {MWh (primary), CO2 (complementary)},
%   estimated in separate runs
% - E_p: first EDWC go-live month for province p
% - l = t - E_p: event time; omit l = -1 as reference
% - 1{p in G}: treated set (Gansu, Ningxia, IM, Guizhou)
% - alpha_p: province FE; delta_t: calendar-month FE
% - X_{p,t}: weather (CDD, HDD, precipitation), holidays
% - beta_l: event-time ATT (Sun--Abraham cohort-interaction-weighted)
% - epsilon_{p,t}: clustered by province
%
% Event window: l in [-12, +18] months; tails binned.
% Estimation: cohort-specific regressions aggregated per
% Sun and Abraham (2021).

\subsubsection{Compute Translation Framework (RQ2)}
% PURPOSE: Present the CPC framework as the translation layer
% that operationalizes RQ2, downstream from RQ1.
%
% CPC DEFINITION:

\begin{equation}
\text{CPC}_{p,t}
= \text{CI}_{p,t} \times \text{PUE}_{p,t}
  \times \frac{1000}{\text{GF/W}_{\text{AI,China}}(t)}
\quad \left[\frac{\text{tCO}_2}{\text{EF}\cdot\text{h}}\right]
\label{eq:cpc}
\end{equation}

% GF/W CONSTRUCTION:

\begin{equation}
\text{GF/W}_{\text{AI,China}}(t)
= \underbrace{\text{GF/W}_{\text{FP64}}(t) \cdot s_{\text{mxp}}(t)
  \cdot \kappa_{\text{global}}}_{\text{GF/W}_{\text{AI,global}}(t)}
  \times \chi_{\text{CN}}(t)
\label{eq:gfw}
\end{equation}

% - CI computed from CO2 and kWh panels (DiD-CPC consistency)
